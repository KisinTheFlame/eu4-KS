# Kang problem
country_event = {
    id = ks_ming_event_t.1
    title = ks_ming_event_t.kang_problem.title
    desc = ks_ming_event_t.kang_problem.description

    is_triggered_only = yes
    
    option = {
        name = ks_ming_event_t.kang_problem.option.tight

        set_country_flag = tight_kang_policy
        if = {
            limit = {
                exists = KAM
                KAM = {
                    is_subject_of = root
                }
                NOT = {
                    KAM = {
                        OR = {
                            vassal_of = root
                            march_of = root
                        }
                    }
                }
            }
            vassalize = KAM
        }
        else_if = {
            limit = {
                KAM = {
                    OR = {
                        vassal_of = root
                        march_of = root
                    }
                }
            }
            KAM = {
                add_liberty_desire = 15
            }
            add_country_modifier = {
                name = ks.ming.modifier.tight_kang_policy
                duration = 7300
            }
        }
        else = {
            kham_area = {
                add_base_tax = 1
            }
        }
    }

    option = {
        name = ks_ming_event_t.kang_problem.option.loose

        set_country_flag = loose_kang_policy
        if = {
            limit = {
                exists = KAM
                KAM = {
                    is_subject_of = root
                }
                NOT = {
                    KAM = {
                        OR = {
                            vassal_of = root
                            march_of = root
                        }
                    }
                }
            }
            create_march = KAM
        }
        else_if = {
            limit = {
                KAM = {
                    OR = {
                        vassal_of = root
                        march_of = root
                    }
                }
            }
            KAM = {
                add_liberty_desire = -15
            }
        }
        else = {
            kham_area = {
                add_nationalism = -10
            }
        }
    }
}
