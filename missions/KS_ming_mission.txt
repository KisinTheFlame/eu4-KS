ks.ming.missions.manchu_affairs = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = MNG
			was_tag = MNG
		}
	}
	has_country_shield = yes
	ks.ming.mission.consolidate_jianzhou = {
		position = 1
		icon = mission_non-western_cavalry_raid
		required_missions = {
		}
		trigger = {
			OR = {
				calc_true_if = {
					all_subject_country = {
						OR = {
							dominant_culture = manchu
							dominant_culture = manchu_new
						}
						is_vassal = no
						has_opinion = {
							who = root
							value = 150
						}
					}
					amount = 2
				}
				AND = {
					jilin_area = {
						type = all
						country_or_non_sovereign_subject_holds = root
					}
					ningguta_area = {
						type = all
						country_or_non_sovereign_subject_holds = root
					}
				}
			}
		}
		effect = {
			custom_tooltip = ks.ming.mission.consolidate_jianzhou.effect.tooltip
			hidden_effect = {
				if = {
					limit = {
						calc_true_if = {
							all_subject_country = {
								OR = {
									dominant_culture = manchu
									dominant_culture = manchu_new
								}
								has_opinion = {
									who = root
									value = 150
								}
							}
							amount = 2
						}
					}
					every_subject_country = {
						limit = {
							OR = {
								dominant_culture = manchu
								dominant_culture = manchu_new
							}
							is_vassal = no
							has_opinion = {
								who = root
								value = 150
							}
						}
						root = {
							vassalize = prev
						}
					}
				}
				else = {
					every_province = {
						limit = {
							region = manchuria_region
							OR = {
								culture = manchu
								culture = manchu_new
							}
						}
						add_permanent_claim = root
					}
				}
			}
		}
	}
	ks.ming.mission.nurgan_regional_military_commission = {
		position = 2
		icon = mission_manchu_warrior
		required_missions = {
			ks.ming.mission.consolidate_jianzhou
		}
		trigger = {
			jilin_area = {
				type = all
				owned_by = root
			}
			ningguta_area = {
				type = all
				owned_by = root
			}
			furdan_area = {
				type = all
				owned_by = root
			}
		}
		effect = {
			country_event = {
				id = ks_ming_event_t.6
			}
		}
	}
}

ks.ming.missions.far_east_affairs = {
	slot = 2
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = MNG
			was_tag = MNG
		}
	}
	has_country_shield = yes
	ks.ming.mission.sweep_out_japanese_pirates = {
		position = 2
		icon = mission_japanese_samurai
		required_missions = {
			ks.ming.mission.navy_of_ming
		}
		trigger = {
			OR = {
				any_country = {
					has_reform = shogunate
					OR = {
						AND = {
							alliance_with = root
							has_opinion = {
								who = root
								value = 150
							}
						}
						AND = {
							NOT = {
								truce_with = root
							}
							japan_region = {
								country_or_non_sovereign_subject_holds = root
							}
						}
						is_subject_of = root
					}
				}
				all_country = {
					NOT = {
						has_reform = shogunate
					}
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = ks.ming.modifier.swept_out_japanese_pirates
				duration = 3650
			}
		}
	}
	ks.ming.mission.our_attitude_to_choson = {
		position = 3
		icon = mission_noble_council
		required_missions = {
			ks.ming.mission.sweep_out_japanese_pirates
		}
		trigger = {
			if = {
				limit = {
					exists = KOR
				}
				OR = {
					is_pleasing_subject_of = {
						subject = KOR
						master = root
					}
					AND = {
						is_rival = KOR
						trade_embargoing = KOR
					}
				}
			}
			else = {
				pyongan_area = {
					type = all
					country_or_non_sovereign_subject_holds = root
				}
				western_korea_area = {
					type = all
					country_or_non_sovereign_subject_holds = root
				}
			}
		}
		effect = {
			if = {
				limit = {
					exists = KOR
				}
				if = {
					limit = {
						KOR = {
							is_subject_of = root
							has_opinion = {
								who = root
								value = 150
							}
						}
					}
					country_event = {
						id = ks_ming_event_t.2
					}
				}
				else_if = {
					limit = {
						is_rival = KOR
						trade_embargoing = KOR
					}
					add_casus_belli = {
						type = cb_vassalize_mission
						target = KOR
						months = 360
					}
				}
				else = {
					custom_tooltip = ks.ming.mission.our_attitude_to_choson.effect.tooltip
				}
			}
			else = {
				korea_region = {
					add_permanent_claim = root
				}
			}
		}
	}
}
